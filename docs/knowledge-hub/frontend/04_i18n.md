## 1. i18n Setup

### Dependencies

```json
{
  "dependencies": {
    "i18next": "^23.0.0",
    "react-i18next": "^14.0.0"
  }
}
```

### Configuration

```typescript
// src/i18n/index.ts
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';

const resources = {
  ar: {
    translation: {
      // Arabic translations
      "welcome": "مرحباً",
      "dashboard": "لوحة التحكم",
    }
  },
  en: {
    translation: {
      // English translations
      "welcome": "Welcome",
      "dashboard": "Dashboard",
    }
  }
};

i18n
  .use(initReactI18next)
  .init({
    resources,
    lng: 'ar', // Default language
    fallbackLng: 'en',
    interpolation: {
      escapeValue: false, // React already escapes
    }
  });

export default i18n;
```

## 3. Using Translations

### useTranslation Hook

```typescript
import { useTranslation } from 'react-i18next';

function DashboardPage() {
  const { t } = useTranslation();
  
  return (
    <div>
      <h1>{t('nav.dashboard')}</h1>
      <button>{t('action.create')}</button>
    </div>
  );
}
```

### With Variables

```typescript
// In translations
"welcome_user": "Welcome, {{name}}"

// In component
const name = 'John';
return <p>{t('welcome_user', { name })}</p>;
// Output: "Welcome, John"
```

### Pluralization

```typescript
// Translations
"items_count": "{{count}} items",
"items_count_one": "{{count}} item",
"items_count_other": "{{count}} items"

// Usage
t('items_count', { count: 1 });    // "1 item"
t('items_count', { count: 5 });    // "5 items"
```

## 4. Language Switching

### Toggle Component

```tsx
import { useTranslation } from 'react-i18next';

function LanguageToggle() {
  const { i18n } = useTranslation();
  
  const toggle = () => {
    const newLang = i18n.language === 'ar' ? 'en' : 'ar';
    i18n.changeLanguage(newLang);
  };
  
  return (
    <button onClick={toggle}>
      {i18n.language === 'ar' ? 'EN' : 'عربي'}
    </button>
  );
}
```

### Persist Selection

```typescript
// Save to localStorage
const changeLanguage = (lang: string) => {
  i18n.changeLanguage(lang);
  localStorage.setItem('language', lang);
};

// Load on init
const savedLang = localStorage.getItem('language');
if (savedLang) {
  i18n.changeLanguage(savedLang);
}
```

## 5. RTL Support

### CSS Direction

```typescript
// Set document direction
useEffect(() => {
  document.documentElement.dir = i18n.language === 'ar' ? 'rtl' : 'ltr';
  document.documentElement.lang = i18n.language;
}, [i18n.language]);
```

### Tailwind RTL Classes

```html
<!-- Use logical properties -->
<div className="ms-4">    <!-- margin-start instead of margin-left -->
<div className="me-4">    <!-- margin-end -->
<div className="ps-4">    <!-- padding-start -->
<div className="text-start">   <!-- text-start instead of text-left -->
```
